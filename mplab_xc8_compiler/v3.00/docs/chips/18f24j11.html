18F24J11 Support Information

XC8-CC #pragma config Usage

#pragma config <setting>=<named value>
For example:
// Watchdog Timer : Enabled
// Stack Overflow/Underflow Reset : Enabled
// Extended Instruction Set : Enabled
// Background Debug : Disabled
#pragma config WDTEN = ON, STVREN = ON, XINST = ON, DEBUG = OFF
#pragma config <setting>=<literal constant>
For example:
// Watchdog Timer : Enabled
// Stack Overflow/Underflow Reset : Enabled
// Extended Instruction Set : Enabled
// Background Debug : Disabled
#pragma config WDTEN = 0x1, STVREN = 0x1, XINST = 0x1, DEBUG = 0x1
#pragma config <register>=<literal constant>
For example:
// Watchdog Timer : Enabled
// Stack Overflow/Underflow Reset : Enabled
// Extended Instruction Set : Enabled
// Background Debug : Disabled
#pragma config CONFIG1L = 0xE1

PIC-AS config Usage

config "<setting>"="<named value>"
For example:
; Watchdog Timer : Enabled
; Stack Overflow/Underflow Reset : Enabled
; Extended Instruction Set : Enabled
; Background Debug : Disabled
config "WDTEN" = "ON", "STVREN" = "ON", "XINST" = "ON", "DEBUG" = "OFF"
config "<setting>"=<literal constant>
For example:
; Watchdog Timer : Enabled
; Stack Overflow/Underflow Reset : Enabled
; Extended Instruction Set : Enabled
; Background Debug : Disabled
config "WDTEN" = 0x1, "STVREN" = 0x1, "XINST" = 0x1, "DEBUG" = 0x1
config "<register>"=<literal constant>
For example:
; Watchdog Timer : Enabled
; Stack Overflow/Underflow Reset : Enabled
; Extended Instruction Set : Enabled
; Background Debug : Disabled
config "CONFIG1L" = 0xE1
