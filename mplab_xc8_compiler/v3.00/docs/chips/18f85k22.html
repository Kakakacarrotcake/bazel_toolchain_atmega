18F85K22 Support Information

XC8-CC #pragma config Usage

#pragma config <setting>=<named value>
For example:
// VREG Sleep Enable bit : Enabled
// LF-INTOSC Low-power Enable bit : LF-INTOSC in High-power mode during Sleep
// SOSC Power Selection and mode Configuration bits : High Power SOSC circuit selected
// Extended Instruction Set : Enabled
#pragma config RETEN = ON, INTOSCSEL = HIGH, SOSCSEL = HIGH, XINST = ON
#pragma config <setting>=<literal constant>
For example:
// VREG Sleep Enable bit : Enabled
// LF-INTOSC Low-power Enable bit : LF-INTOSC in High-power mode during Sleep
// SOSC Power Selection and mode Configuration bits : High Power SOSC circuit selected
// Extended Instruction Set : Enabled
#pragma config RETEN = 0x1, INTOSCSEL = 0x1, SOSCSEL = 0x3, XINST = 0x1
#pragma config <register>=<literal constant>
For example:
// VREG Sleep Enable bit : Enabled
// LF-INTOSC Low-power Enable bit : LF-INTOSC in High-power mode during Sleep
// SOSC Power Selection and mode Configuration bits : High Power SOSC circuit selected
// Extended Instruction Set : Enabled
#pragma config CONFIG1L = 0x5D
For example:
// IDLOC @ 0x200000
#pragma config IDLOC0 = 0x255
