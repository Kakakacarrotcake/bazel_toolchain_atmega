18F86J93 Support Information

XC8-CC #pragma config Usage

#pragma config <setting>=<named value>
For example:
// Watchdog Timer : Enabled
// Stack Overflow Reset : Enabled
// Extended Instruction Set Enable bit : Enabled
#pragma config WDTEN = ON, STVREN = ON, XINST = ON
#pragma config <setting>=<literal constant>
For example:
// Watchdog Timer : Enabled
// Stack Overflow Reset : Enabled
// Extended Instruction Set Enable bit : Enabled
#pragma config WDTEN = 0x1, STVREN = 0x1, XINST = 0x1
#pragma config <register>=<literal constant>
For example:
// Watchdog Timer : Enabled
// Stack Overflow Reset : Enabled
// Extended Instruction Set Enable bit : Enabled
#pragma config CONFIG1L = 0xE1

PIC-AS config Usage

config "<setting>"="<named value>"
For example:
; Watchdog Timer : Enabled
; Stack Overflow Reset : Enabled
; Extended Instruction Set Enable bit : Enabled
config "WDTEN" = "ON", "STVREN" = "ON", "XINST" = "ON"
config "<setting>"=<literal constant>
For example:
; Watchdog Timer : Enabled
; Stack Overflow Reset : Enabled
; Extended Instruction Set Enable bit : Enabled
config "WDTEN" = 0x1, "STVREN" = 0x1, "XINST" = 0x1
config "<register>"=<literal constant>
For example:
; Watchdog Timer : Enabled
; Stack Overflow Reset : Enabled
; Extended Instruction Set Enable bit : Enabled
config "CONFIG1L" = 0xE1
