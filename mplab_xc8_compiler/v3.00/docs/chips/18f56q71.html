For example:
// External Oscillator Selection : EC (external clock) above 8 MHz

#pragma config FEXTOSC = ECH, RSTOSC = EXTOSC
#pragma config <setting>=<literal constant>
For example:
// External Oscillator Selection : EC (external clock) above 8 MHz

#pragma config FEXTOSC = 0x7, RSTOSC = 0x7
