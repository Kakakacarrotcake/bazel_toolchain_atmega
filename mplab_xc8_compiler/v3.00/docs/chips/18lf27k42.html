For example:
// External Oscillator Selection : EC (external clock) above 8 MHz; PFM set to high power

#pragma config FEXTOSC = ECH, RSTOSC = EXTOSC

HFINTOSC_64MHZ HFINTOSC with HFFRQ = 64 MHz and CDIV = 1:1
