18F26J50 Support Information

XC8-CC #pragma config Usage

#pragma config <setting>=<named value>
For example:
// Watchdog Timer : Enabled
// PLL Prescaler Selection bits : No prescale (4 MHz oscillator input drives PLL directly)
// Stack Overflow/Underflow Reset : Enabled
// Extended Instruction Set : Enabled
// Background Debug : Disabled
#pragma config WDTEN = ON, PLLDIV = 1, STVREN = ON, XINST = ON, DEBUG = OFF
#pragma config <setting>=<literal constant>
For example:
// Watchdog Timer : Enabled
// PLL Prescaler Selection bits : No prescale (4 MHz oscillator input drives PLL directly)
// Stack Overflow/Underflow Reset : Enabled
// Extended Instruction Set : Enabled
// Background Debug : Disabled
#pragma config WDTEN = 0x1, PLLDIV = 0x7, STVREN = 0x1, XINST = 0x1, DEBUG = 0x1
#pragma config <register>=<literal constant>
For example:
// Watchdog Timer : Enabled
// PLL Prescaler Selection bits : No prescale (4 MHz oscillator input drives PLL directly)
// Stack Overflow/Underflow Reset : Enabled
// Extended Instruction Set : Enabled
// Background Debug : Disabled
#pragma config CONFIG1L = 0xEF
